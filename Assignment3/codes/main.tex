\section*{Problem: GATE 2002(CS)-Q.41}
Four fair coins are tossed simultaneously. The probability that at least one head and one tail turn up is
\begin{enumerate}[label=\alph*)]
    \item $\frac{1}{16}$\\
    \item $\frac{1}{8}$\\
    \item $\frac{7}{8}$\\
    \item $\frac{15}{16}$
\end{enumerate}
\section*{solution}
\\Let Y $\in$ $\{0,1\}$ be the random variable such that 1 represents occurrence of tail,0 represents occurrence of head when coin is tossed.
\begin{table}[ht]
\caption{Probability distribution for values of Y}
\begin{center}
    \begin{tabular}{|c|c|}
    \hline
    Y & P(Y)\\
    \hline
    1 & $\frac{1}{2}$ \\
    \hline
    0 & $\frac{1}{2}$\\
    \hline
    \end{tabular}
\end{center} 
\end{table}\\
Let the discrete random variable x is following bernoulli distribution with parameters $n,p$ where $n=4,p=\text{Pr}(Y=0)=\frac{1}{2}$ and $q=\text{Pr}(Y=1)=\frac{1}{2}$\\
Probability mass function of x successes in n trials is given by
\begin{align}
    \text{Pr}(X=x)=\Mycomb{x} p^{x}q^{n-x}
\end{align}
\begin{align}
 \sum_{j=1}^{3}\text{Pr}(X=j)=\Mycomb[4]{1}(1/2)^{1}(1/2)^{3}+\Mycomb[4]{2}(1/2)^{2}(1/2)^{2}\\+\Mycomb[4]{3}(1/2)^{3}(1/2)^{4}
\end{align}
\begin{align}
    \sum_{j=1}^{3}\text{Pr}(X=j)=\frac{7}{8}
\end{align}
 \text{The required probability is } \fbox{$\frac{7}{8}$}\\
\textbf{$\therefore$ Option C is true}
